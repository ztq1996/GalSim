570  inline void intGKP(
571  const UF& func, IntRegion<typename UF::result_type>& reg,
572  const typename UF::result_type relerr,
573  const typename UF::result_type abserr)
574  {

576  const T eps = std::numeric_limits<T>::epsilon();
577

579 
580  assert(abserr >= 0.);
581  assert(relerr > 0.);
582 
583  // Check for early exit:
584  if (reg.left() == reg.right()) {
585  integ_dbg2<<"left == right, so integral is trivially 0.\n";
586  reg.setArea(0.,0.);
587  return;
588  }
589 
590  // Perform the first integration
591  bool done = intGKPNA(func, reg, relerr, abserr);
592  if (done) {
593  integ_dbg2<<"GKPNA suceeded, so we're done.\n";
594  return;
595  }
596

598  integ_dbg2<<"Intial range = "<<reg.left()<<".."<<reg.right()<<std::endl;
599

601  T roundoff_type2 = 0.;

603 
604  // Keep track of all subdivision in a priority_queue.
605  // The top() is always the largest value, and pop() removes it.
606  // We define < and > for IntRegoins such that the "largest" is the one
607  // with the largest current error estimate. This is the next one to be split
608  // if we need to split further.
609  std::priority_queue<IntRegion<T>,std::vector<IntRegion<T> > > allregions;
610  allregions.push(reg);
611  T finalarea = reg.getArea();
612  T finalerr = reg.getErr();

614  assert(finalerr > tolerance);

620  IntRegion<T> parent = allregions.top();

624  integ_dbg2<<"parent area = "<<parent.getArea();
625  integ_dbg2<<" +- "<<parent.getErr()<<std::endl;

630

632  T factor = 3*children.size()*finalerr/tolerance;
633  T newabserr = std::abs(parent.getErr()/factor);
634  T newrelerr = newabserr/std::abs(parent.getArea());
635  integ_dbg2<<"New abserr,rel = "<<newabserr<<','<<newrelerr;
636  integ_dbg2<<" ("<<children.size()<<" children)\n";
