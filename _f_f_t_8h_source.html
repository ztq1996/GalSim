119  // Handle the FFTW3 memory allocation, which assured 16-bit alignment for SSE usage.
120  // FFTW3 now states that C++ complex<double> will be bit-compatible with
121  // the fftw_complex type. So all interfaces will be through std::complex<double>
122  // And the fftw real type is now just double.
123  template <typename T>
124  class FFTW_Array
125  {
126  typedef typename FFTW_Traits<T>::fftw_type fftw_type;
127  public:
128  FFTW_Array() : _n(0), _p(0) {}
129 
130  FFTW_Array(size_t n) : _n(0), _p(0)
131  { resize(n); }
132 
133  FFTW_Array(size_t n, T val) : _n(0), _p(0)
134  {
135  resize(n);
136  fill(val);
137  }
138 
139  FFTW_Array(const FFTW_Array<T>& rhs) : _n(0), _p(0)
140  {
141  resize(rhs._n);
142  for (size_t i=0; i<_n; ++i) _p[i] = rhs._p[i];
143  }
144 
145  FFTW_Array<T>& operator=(const FFTW_Array<T>& rhs)
146  {
147  if (this != &rhs) {
148  resize(rhs._n);
149  for (size_t i=0; i<_n; ++i) _p[i] = rhs._p[i];
150  }
151  return *this;
152  }
153 
154  ~FFTW_Array() {}
155 
156  void resize(size_t n)
157  {
158  if (_n != n) {
159  _n = n;
160  _array.resize(n);
161  _p = _array.get();
162  }
163  }
164 
165  void fill(T val)
166  {
167  for (size_t i=0; i<_n; ++i) _p[i] = val;
168  }
169 
170  size_t size() { return _n; }
171 
172  T* get() { return _p; }
173  const T* get() const { return _p; }
174 
175  fftw_type* get_fftw()
176  { return reinterpret_cast<fftw_type*>(_p); }
177  const fftw_type* get_fftw() const
178  { return reinterpret_cast<const fftw_type*>(_p); }
179 
180  T& operator[](size_t i) { return _p[i]; }
181  const T& operator[](size_t i) const { return _p[i]; }

184  size_t _n;

521  template <class T>
522  void KTable::fill(const T& f)
523  {
524  clearCache(); // invalidate any stored interpolations
525  std::complex<double>* zptr=_array.get();
526  double kx, ky;
527  for (int iy=0; iy< _No2; iy++) {
528  ky = iy*_dk;
529  for (int ix=0; ix< _No2+1 ; ix++) {
530  kx = ix*_dk;
531  *(zptr++) = f(kx,ky);
532  }
533  }
534  // wrap to the negative ky's
535  for (int iy=-_No2; iy< 0; iy++) {
536  ky = iy*_dk;
537  for (int ix=0; ix< _No2+1 ; ix++) {
538  kx = ix*_dk;
539  *(zptr++) = f(kx,ky);
540  }
541  }
542  return;
543  }
