82  inline double getBesselRoot0(int s)
83  {
84  if (s <= 0)
85  throw std::runtime_error("s must be > 0");
86  if (s <= n_roots_j0) return root_j0[s-1];
87  else {
88  // Above this value, the asymptotic formula from Abramowitz and Stegun 9.5.12
89  // is accurate to better than 1.e-16, so essentially exact.
90  // b - (m-1)/8b - 4(m-1)(7m-31)/3(8b)^3 - 32(m-1)(83m^2-982m+3779)/15(8b)^5
91  // - 64(m-1)(6949m^3-153855m^2+1585743m-6277237)/105(8b)^7
92  // where m = 4nu^2 and b = (s + nu/2 - 1/4)pi.
93  //
94  // For nu = 0, this simplifies considerably to:
95  // b + 1/8b - 4*31/3(8b)^3 + 32*3779/15(8b)^5 - 64*6277237/105(8b)^7
96  double b = (s - 0.25) * M_PI;
97  double temp = 0.125/b;
98  double inv8bsq = temp*temp;
99 
100  // From here on, b will be the running total, not the original b, and temp will be
101  // the part of each coefficient that is in common with the next term.
102  b += temp;
103 
104  temp *= (4./3.) * inv8bsq;
105  b -= 31. * temp;
106 
107  temp *= (8./5.) * inv8bsq;
108  b += 3779. * temp;
109 
110  temp *= (2./7.) * inv8bsq;
111  b -= 6277237. * temp;
112 
113  return b;
114  }
115  }
