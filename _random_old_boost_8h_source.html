41  class UniformDeviate
42  // Uniform wrapper is given arguments to map RNG's to [0.,1.) interval.
43  {
44  public:
45  UniformDeviate(): urd(0.,1.) { seedtime(); } // seed with time
46  UniformDeviate(const long lseed): urng(lseed), urd(0.,1.) {} //seed with specific number
47  double operator() () { return urd(urng); }
48  operator double() { return urd(urng); }
49  void seed() { seedtime(); }
50  void seed(const long lseed) { urng.seed(lseed); }
51 
52  private:
53  OurURNG urng;
54  boost::uniform_real<> urd;
55  void seedtime()
56  {
57  struct timeval tp;
58  gettimeofday(&tp,NULL);
59  urng.seed(tp.tv_usec);
60  }
61  // Hide copy and assignment so users do not create duplicate (correlated!) RNG's:
62  UniformDeviate(const UniformDeviate& rhs) {}
63  void operator=(const UniformDeviate& rhs) {}
64 
65  // make friends able to see the RNG without the distribution wrapper:
66  friend class GaussianDeviate;
67  friend class PoissonDeviate;
68  friend class BinomialDeviate;
69 
70  };

72  // A Gaussian deviate (unit variance, zero-mean by default).
73  // Wraps the Boost.Random normal_distribution so that
74  // the parent UniformDeviate is given once at construction, and copy/assignment are hidden.
75  class GaussianDeviate
76  {
77  typedef boost::variate_generator<OurURNG&,
78  boost::normal_distribution<> > GaussGenerator;
79  public:
80  GaussianDeviate(UniformDeviate& u_, double mean=0., double sigma=1.) :
81  u(u_),
82  normal(new boost::normal_distribution<>(mean,sigma)),
83  gen(new GaussGenerator(u.urng,*normal)) {}
84  ~GaussianDeviate() {if (gen) delete gen; if (normal) delete normal;}
85  double operator() () { return (*gen)(); }
86  operator double() { return (*gen)();}
87  double getMean() {return normal->mean();}
88  double getSigma() {return normal->sigma();}
89  void setMean(double mean) {
90  boost::normal_distribution<>* normal2 =
91  new boost::normal_distribution<>(mean,normal->sigma());
92  delete normal;
93  delete gen;
94  normal = normal2;
95  gen = new GaussGenerator(u.urng,*normal);
96  }
97  void setSigma(double sigma) {
98  boost::normal_distribution<>* normal2 =
99  new boost::normal_distribution<>(normal->mean(),sigma);
100  delete normal;
101  delete gen;
102  normal = normal2;
103  gen = new GaussGenerator(u.urng,*normal);
104  }
105  private:
106  UniformDeviate& u;
107  boost::normal_distribution<>* normal;
108  GaussGenerator* gen;
109  // Hide:
110  GaussianDeviate(const GaussianDeviate& rhs): u(rhs.u), gen(0), normal(0) {}
111  void operator=(const GaussianDeviate& rhs) {}
112  };
