590  template <class U>
591  void copyFrom(const BaseImage<U>& rhs) const
592  {
593  if (!this->getBounds().isSameShapeAs(rhs.getBounds()))
594  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
595  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
596  iterator it1 = rowBegin(y);
597  const iterator ee = rowEnd(y);
598  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
599  while (it1 != ee) *(it1++) = T(*(it2++));
600  }
601  }
