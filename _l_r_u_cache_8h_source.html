123  boost::shared_ptr<Value> get(const Key& key)
124  {
125  assert(_entries.size() == _cache.size());
126  MapIter iter = _cache.find(key);
127  if (iter != _cache.end()) {
128  // Item is cached.
129  // Move it to the front of the list.
130  if (iter != _cache.begin())
131  _entries.splice(_entries.begin(), _entries, iter->second);
132  // Return the item's value
133  assert(_entries.size() == _cache.size());
134  return iter->second->second;
135  } else {
136  // Item is not cached.
137  // Make a new one.
138  boost::shared_ptr<Value> value(LRUCacheHelper<Value,Key>::NewValue(key));
139  // Remove items from the cache as necessary.
140  while (_entries.size() >= _nmax) {
141  bool erased = _cache.erase(_entries.back().first);
142  assert(erased);
143  _entries.pop_back();
144  }
145  // Add the new value to the front.
146  _entries.push_front(Entry(key,value));
147  // Also put it in the cache
148  _cache[key] = _entries.begin();
149  // Return the new value
150  assert(_entries.size() == _cache.size());
151  return value;
152  }
153  }
154 
155  private:
156 
157  size_t _nmax;

159  typedef std::pair<Key, boost::shared_ptr<Value> > Entry;

161 
162  typedef typename std::list<Entry>::iterator ListIter;
163  std::map<Key, ListIter> _cache;
