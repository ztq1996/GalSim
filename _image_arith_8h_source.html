70  template <typename T, typename Op>
71  Op for_each_pixel(const ImageView<T>& image, Op f)
72  {
73  // Note: all of these functions have this guard to make sure we don't
74  // try to access the memory if the image is in an undefined state.
75  if (image.getData()) {
76  if (image.isContiguous()) {
77  f = std::for_each(image.rowBegin(image.getYMin()), image.rowEnd(image.getYMax()), f);
78  } else {
79  for (int i = image.getYMin(); i <= image.getYMax(); i++)
80  f = std::for_each(image.rowBegin(i), image.rowEnd(i), f);
81  }
82  }
83  return f;
84  }

89  template <typename T, typename Op>
90  Op for_each_pixel(const ImageView<T>& image, const Bounds<int>& bounds, Op f)
91  {
92  if (image.getData()) {
93  if (!image.getBounds().includes(bounds))
94  throw ImageError("for_each_pixel range exceeds image range");
95 
96  if (image.getBounds() == bounds) return for_each_pixel(image,f);
97 
98  for (int i = bounds.getYMin(); i <= bounds.getYMax(); i++)
99  f = std::for_each(image.getIter(bounds.getXMin(),i),
100  image.getIter(bounds.getXMax()+1,i), f);
101  }
102  return f;
103  }

108  template <typename T, typename Op>
109  Op transform_pixel(const ImageView<T>& image, Op f)
110  {
111  if (image.getData()) {
112  typedef typename ImageView<T>::iterator Iter;
113  if (image.isContiguous()) {
114  const Iter ee = image.rowEnd(image.getYMax());
115  for (Iter it = image.rowBegin(image.getYMin()); it != ee; ++it)
116  *it = f(*it);
117  } else {
118  for (int y = image.getYMin(); y <= image.getYMax(); ++y) {
119  const Iter ee = image.rowEnd(y);
120  for (Iter it = image.rowBegin(y); it != ee; ++it)
121  *it = f(*it);
122  }
123  }
124  }
125  return f;
126  }

233  // Assign function of 2 images to 1st
234  template <typename T1, typename T2, typename Op>
235  Op transform_pixel(const ImageView<T1>& image1, const BaseImage<T2>& image2, Op f)
236  {
237  if (image1.getData()) {
238  typedef typename ImageView<T1>::iterator Iter1;
239  typedef typename BaseImage<T2>::const_iterator Iter2;
240 
241  if (!image1.getBounds().isSameShapeAs(image2.getBounds()))
242  throw ImageError("transform_pixel image bounds are not same shape");
243 
244  int y2 = image2.getYMin();
245  for (int y = image1.getYMin(); y <= image1.getYMax(); ++y, ++y2) {
246  Iter2 it2 = image2.rowBegin(y2);
247  const Iter1 ee = image1.rowEnd(y);
248  for (Iter1 it1 = image1.rowBegin(y); it1 != ee; ++it1, ++it2)
249  *it1 = f(*it1,*it2);
250  }
251  }
252  return f;
253  }

255  // Assign function of Img2 & Img3 to Img1
256  template <typename T1, typename T2, typename T3, typename Op>
257  Op transform_pixel(
258  const ImageView<T1>& image1,
259  const BaseImage<T2>& image2,
260  const BaseImage<T3>& image3,
261  Op f)
262  {
263  if (image1.getData()) {
264  typedef typename ImageView<T1>::iterator Iter1;
265  typedef typename BaseImage<T2>::const_iterator Iter2;
266  typedef typename BaseImage<T3>::const_iterator Iter3;
267 
268  if (!image1.getBounds().isSameShapeAs(image2.getBounds()))
269  throw ImageError("transform_pixel image1, image2 bounds are not same shape");
270  if (!image1.getBounds().isSameShapeAs(image3.getBounds()))
271  throw ImageError("transform_pixel image1, image3 bounds are not same shape");
272 
273  int y2 = image2.getYMin();
274  int y3 = image3.getYMin();
275  for (int y = image1.getYMin(); y <= image1.getYMax(); ++y, ++y2, ++y3) {
276  Iter2 it2 = image2.rowBegin(y2);
277  Iter3 it3 = image3.rowBegin(y3);
278  const Iter1 ee = image1.rowEnd(y);
279  for (Iter1 it1 = image1.rowBegin(y); it1 != ee; ++it1, ++it2, ++it3)
280  *it1 = f(*it2,*it3);
281  }
282  }
283  return f;
284  }

286  // Assign function of 2 images to 1st over bounds
287  // Note that for this one, the two images do not have to be the same shape.
288  // They just both have to include the given bounds.
289  // Only that portion of each image is used for the calculation.
290  template <typename T1, typename T2, typename Op>
291  Op transform_pixel(
292  const ImageView<T1>& image1,
293  const BaseImage<T2>& image2,
294  const Bounds<int>& bounds,
295  Op f)
296  {
297  if (image1.getData()) {
298  typedef typename ImageView<T1>::iterator Iter1;
299  typedef typename Image<T2>::iterator Iter2;
300  if (!image1.getBounds().includes(bounds) || !image2.getBounds().includes(bounds))
301  throw ImageError("transform_pixel range exceeds image range");
302 
303  if (image1.getBounds() == bounds && image2.getBounds() == bounds)
304  return transform_pixel(image1,image2,f);
305 
306  for (int y = bounds.getYMin(); y <= bounds.getYMax(); ++y) {
307  const Iter1 ee = image1.getIter(bounds.getXMax()+1,y);
308  Iter2 it2 = image2.getIter(bounds.getXMin(),y);
309  for (Iter1 it1 = image1.getIter(bounds.getXMin(),y); it1 != ee; ++it1, ++it2)
310  *it1 = f(*it1,*it2);
311  }
312  }
313  return f;
314  }
