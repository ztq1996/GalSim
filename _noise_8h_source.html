282  template <typename T>
283  void applyTo(ImageView<T> data)
284  {
285  // Above this many e's, assume Poisson distribution == Gaussian
286  // The Gaussian deviate is about 20% faster than Poisson, and for high N
287  // they are virtually identical.
288  const double MAX_POISSON=1.e5;
289  // Typedef for image row iterable
290  typedef typename ImageView<T>::iterator ImIter;
291 
292  data += T(_sky_level);
293 
294  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
295  GaussianDeviate gd(*_rng, 0., 1.);
296  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
297  ImIter ee = data.rowEnd(y);
298  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
299  if (*it <= 0.) continue;
300  if (*it < MAX_POISSON) {
301  pd.setMean(*it);
302  *it = T(pd());
303  } else {
304  gd.setSigma(sqrt(*it));
305  *it = T(*it + gd());
306  }
307  }
308  }
309 
310  data -= T(_sky_level);
311  }

444  void applyTo(ImageView<T> data)
445  {
446  // Above this many e's, assume Poisson distribution == Gaussian
447  // The Gaussian deviate is about 20% faster than Poisson, and for high N
448  // they are virtually identical.
449  const double MAX_POISSON=1.e5;
450  // Typedef for image row iterable
451  typedef typename ImageView<T>::iterator ImIter;
452 
453  data += T(_sky_level);
454 
455  // Add the Poisson noise first:
456  if (_gain > 0.) {
457  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
458  GaussianDeviate gd(*_rng, 0., 1.);
459  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
460  ImIter ee = data.rowEnd(y);
461  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
462  if (*it <= 0.) continue;
463  double electrons = *it * _gain;
464  if (electrons < MAX_POISSON) {
465  pd.setMean(electrons);
466  *it = T(pd() / _gain);
467  } else {
468  gd.setSigma(sqrt(electrons)/_gain);
469  *it = T(*it + gd());
470  }
471  }
472  }
473  }
474

476  if (_read_noise > 0.) {
477  GaussianDeviate gd(*_rng, 0., _read_noise / (_gain > 0. ? _gain : 1.));

479  ImIter ee = data.rowEnd(y);
480  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
481  *it = T(*it + gd());
482  }

501  {
502  // Typedef for image row iterable
503  typedef typename ImageView<T>::iterator ImIter;
504  assert(data.getBounds() == var.getBounds());
505  // Fill with the (constant) Gaussian contribution to variance
506  if (_read_noise > 0.) {
507  double sigma = _read_noise / (_gain > 0. ? _gain : 1.);
508  var.fill(sigma * sigma);
509  }
510  // Add the Poisson variance:
511  if (_gain > 0.) {
512  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
513  ImIter ee = data.rowEnd(y);
514  ImIter it2 = var.rowBegin(y);
515  for (ImIter it = data.rowBegin(y); it != ee; ++it, ++it2) {
516  if (*it > 0.) *it2 += (*it + _sky_level) / _gain;
517  }
518  }
